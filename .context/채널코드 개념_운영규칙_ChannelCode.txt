CREATE TABLE profiles (

  id BIGSERIAL PRIMARY KEY,

  userId BIGINT NOT NULL,

  profileType TEXT NOT NULL
  CHECK(profileType IN ('GENERAL','BUSINESS')),

  baseCode TEXT NOT NULL
  CHECK(length(baseCode)=12),

BEFORE INSERT ON profiles
BEGIN

CASE
  WHEN substr(NEW.channelCode,2) != NEW.baseCode
  THEN RAISE(ABORT,'channelCode mismatch')

  WHEN NEW.profileType='GENERAL'
   AND substr(NEW.channelCode,1,1)!='A'
  THEN RAISE(ABORT,'invalid prefix')

  WHEN NEW.profileType='BUSINESS'
   AND substr(NEW.channelCode,1,1)!='B'
  THEN RAISE(ABORT,'invalid prefix')
END;
