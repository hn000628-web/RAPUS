// NOTE : 개발용 DB(dev.sqlite) 절대 사용 금지, 경로 강제 절대값 지정, 자동 리셋/삭제 금지

import fs from 'fs';
import path from 'path';
import Database from 'better-sqlite3';
import * as bcrypt from 'bcrypt';

export async function initDatabase() {
  console.log('===== PRODUCTION DB INIT START =====');

  // ==================================================
  // DB 경로 강제 절대값 지정
  // (개발용 dev.sqlite 사용/경로 지정금지)
  // ==================================================
  const dbDir = path.resolve(__dirname, '../../data');
  if (!fs.existsSync(dbDir)) fs.mkdirSync(dbDir, { recursive: true });
  const dbPath = path.join(dbDir, 'prod.sqlite'); // 절대값 지정
  console.log('DB PATH FIXED →', dbPath);

  // ==================================================
  // 운영환경 자동 DROP/RESET 금지
  // ==================================================
  // DB_RESET 환경변수나 DROP TABLE 절대 실행 금지
  const db = new Database(dbPath);

  // ==================================================
  // USERS TABLE
  // ==================================================
  db.exec(`
    CREATE TABLE IF NOT EXISTS users(
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      email TEXT NOT NULL UNIQUE,
      passwordHash TEXT NOT NULL,
      phone TEXT DEFAULT '010-1234-1234',
      displayName TEXT,
      accountType TEXT DEFAULT 'USER',
      status TEXT CHECK(status IN('ACTIVE','INACTIVE')) DEFAULT 'ACTIVE',
      type TEXT DEFAULT 'normal',
      createdAt TEXT DEFAULT CURRENT_TIMESTAMP
    )
  `);

  const testPassword = bcrypt.hashSync('1234', 10);
  const insertUser = db.prepare(`
    INSERT OR IGNORE INTO users(email,passwordHash,displayName,phone,accountType,status,type)
    VALUES(?,?,?,?,?,?,?)
  `);

  // ==================================================
  // 2 TEST ACCOUNTS (AI 임의 수정 금지)
  // ==================================================
  const userId1 = insertUser.run('test1@prod.com', testPassword, '테스트유저1', '010-1234-1234', 'USER', 'ACTIVE', 'normal').lastInsertRowid;
  const userId2 = insertUser.run('test2@prod.com', testPassword, '테스트유저2', '010-1234-1234', 'USER', 'ACTIVE', 'normal').lastInsertRowid;

  // ==================================================
  // PROFILES TABLE
  // ==================================================
  db.exec(`
    CREATE TABLE IF NOT EXISTS profiles(
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      userId INTEGER NOT NULL,
      profileType TEXT NOT NULL CHECK(profileType IN('GENERAL','BUSINESS')),
      displayName TEXT,
      channelCode TEXT NOT NULL UNIQUE,
      channelURL TEXT,
      contactPhone TEXT DEFAULT '010-1234-1234',
      profilePhotoId INTEGER,
      activityRegionId INTEGER,
      feedRegionId INTEGER,
      createdAt TEXT DEFAULT CURRENT_TIMESTAMP,
      updatedAt TEXT DEFAULT CURRENT_TIMESTAMP,
      FOREIGN KEY(userId) REFERENCES users(id)
    )
  `);

  const insertProfile = db.prepare(`
    INSERT OR IGNORE INTO profiles(userId,profileType,displayName,channelCode,channelURL,contactPhone)
    VALUES(?,?,?,?,?,?)
  `);

  // ==================================================
  // 4 PROFILES: 2 GENERAL, 2 BUSINESS (1:1 per TEST ACCOUNT)
  // ※ 테스트 프로필: 채널코드/폰번호 AI 임의 수정 금지
  // ==================================================
  insertProfile.run(userId1,'GENERAL','test1@prod.com_general','A1B2C3D4E5F6','xxx.com/@A1B2C3D4E5F6','010-1234-1234');
  insertProfile.run(userId1,'BUSINESS','test1@prod.com_business','9K2M8P4Q7R1T','xxx.com/@9K2M8P4Q7R1T','010-1234-1234');
  insertProfile.run(userId2,'GENERAL','test2@prod.com_general','Z8X7C6V5B4N3','xxx.com/@Z8X7C6V5B4N3','010-1234-1234');
  insertProfile.run(userId2,'BUSINESS','test2@prod.com_business','Q2W4E6R8T0Y1','xxx.com/@Q2W4E6R8T0Y1','010-1234-1234');

  // ==================================================
  // PROFILE SESSIONS TABLE
  // ==================================================
  db.exec(`
    CREATE TABLE IF NOT EXISTS profile_sessions(
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      userId INTEGER NOT NULL,
      profileId INTEGER NOT NULL,
      profileType TEXT NOT NULL CHECK(profileType IN('GENERAL','BUSINESS')),
      channelCode TEXT NOT NULL,
      deviceId TEXT,
      status TEXT DEFAULT 'ACTIVE',
      loginAt TEXT DEFAULT CURRENT_TIMESTAMP,
      lastSeenAt TEXT DEFAULT CURRENT_TIMESTAMP,
      FOREIGN KEY(userId) REFERENCES users(id),
      FOREIGN KEY(profileId) REFERENCES profiles(id),
      FOREIGN KEY(channelCode) REFERENCES profiles(channelCode)
    )
  `);

  console.log('===== PRODUCTION DB INIT COMPLETE =====');
}

// ==================================================
// ENTRY POINT (단일 실행 보장, import 시 중복 실행 방지)
// ==================================================
if (require.main === module) {
  initDatabase()
    .then(()=>{ console.log('PRODUCTION DB INIT DONE'); process.exit(0); })
    .catch(err=>{ console.error('PRODUCTION DB INIT FAILED',err); process.exit(1); });
}

// SECTION 01 : CHANNEL REGISTRY (IDENTITY ROOT)

TABLE : channels

ROLE :
모든 channel identity root
profile 과 분리된 identity 관리

필수 이유 :

profile delete ≠ channel delete
ownership 변경 대응
채널 상태 관리


COLUMNS :

channelCode TEXT PRIMARY KEY

channelType TEXT NOT NULL
CHECK(channelType IN('PERSONAL','BUSINESS'))

status TEXT NOT NULL DEFAULT 'ACTIVE'
CHECK(status IN(
'ACTIVE',
'SUSPENDED',
'BANNED',
'DELETED'
))

createdAt TEXT DEFAULT CURRENT_TIMESTAMP

deletedAt TEXT



INDEX :

INDEX idx_channels_status(status)



// SECTION 02 : BLOCK SYSTEM (SNS SAFETY CORE)

TABLE : blocks

ROLE :
SNS 차단 시스템

COLUMNS :

id INTEGER PRIMARY KEY AUTOINCREMENT

blockerChannelCode TEXT NOT NULL

blockedChannelCode TEXT NOT NULL

reason TEXT

createdAt TEXT DEFAULT CURRENT_TIMESTAMP

UNIQUE(blockerChannelCode,blockedChannelCode)



INDEX :

INDEX idx_blocks_blocker(blockerChannelCode)

INDEX idx_blocks_blocked(blockedChannelCode)



// SECTION 03 : REPORT SYSTEM (MODERATION CORE)

TABLE : reports

ROLE :
신고 시스템

COLUMNS :

id INTEGER PRIMARY KEY AUTOINCREMENT

reportId TEXT UNIQUE

reporterChannelCode TEXT NOT NULL

targetChannelCode TEXT

targetPostId TEXT

targetCommentId TEXT

reason TEXT

status TEXT DEFAULT 'OPEN'
CHECK(status IN(
'OPEN',
'REVIEWING',
'RESOLVED',
'REJECTED'
))

createdAt TEXT DEFAULT CURRENT_TIMESTAMP

updatedAt TEXT



INDEX :

INDEX idx_reports_reporter(reporterChannelCode)

INDEX idx_reports_status(status)



// SECTION 04 : DIRECT MESSAGE (DM CORE)

TABLE : messages

ROLE :
채팅 시스템

COLUMNS :

id INTEGER PRIMARY KEY AUTOINCREMENT

messageId TEXT UNIQUE

senderChannelCode TEXT NOT NULL

receiverChannelCode TEXT NOT NULL

content TEXT

messageType TEXT DEFAULT 'TEXT'

CHECK(messageType IN(
'TEXT',
'IMAGE',
'SYSTEM'
))

status TEXT DEFAULT 'SENT'

CHECK(status IN(
'SENT',
'DELIVERED',
'READ',
'DELETED'
))

createdAt TEXT DEFAULT CURRENT_TIMESTAMP

updatedAt TEXT



INDEX :

INDEX idx_messages_sender(senderChannelCode)

INDEX idx_messages_receiver(receiverChannelCode)

INDEX idx_messages_created(createdAt)



// SECTION 05 : WALLET (PAYMENT CORE SUPPORT)

TABLE : wallets

ROLE :
결제 잔액 관리

COLUMNS :

id INTEGER PRIMARY KEY AUTOINCREMENT

walletId TEXT UNIQUE

channelCode TEXT NOT NULL

balance INTEGER DEFAULT 0

currency TEXT DEFAULT 'KRW'

status TEXT DEFAULT 'ACTIVE'

CHECK(status IN(
'ACTIVE',
'LOCKED',
'CLOSED'
))

createdAt TEXT DEFAULT CURRENT_TIMESTAMP

updatedAt TEXT



INDEX :

INDEX idx_wallet_channel(channelCode)
